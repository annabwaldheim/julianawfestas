CREATE TABLE `julianawfestas`.`cliente` (`CPF` CHAR(11) NOT NULL , `nome` VARCHAR(50) NOT NULL , `endereco` VARCHAR(150) NOT NULL , `cidade` VARCHAR(50) NOT NULL , `dataNasc` DATE NOT NULL , `telefone` CHAR(11) NOT NULL , PRIMARY KEY (`CPF`)) ENGINE = InnoDB;
CREATE TABLE `julianawfestas`.`login` ( `email` VARCHAR(255) NOT NULL, `senha` VARCHAR(255) NOT NULL, `cpf` CHAR(11) NOT NULL, PRIMARY KEY (`email`), FOREIGN KEY (`cpf`) REFERENCES `cliente`(`cpf`) ON DELETE CASCADE ) ENGINE = InnoDB;
